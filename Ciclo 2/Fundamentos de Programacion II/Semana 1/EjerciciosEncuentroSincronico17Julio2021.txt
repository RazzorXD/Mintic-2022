package clientes;

import java.util.Scanner;



/**
 *
 * @author RamiroJose
 */
public class Clientes {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        
        // Declarar constructor para captura de datos
       Scanner entrarDatos=new Scanner(System.in);
             
        String[] nombreClientes=new String[2];
        String[] apellidosClientes=new String[2];
        boolean visualizar=true;
        int j=0;
        
        for (int i=0;i<=1;i++){
            System.out.println("Cual es el nombre "+"["+i+"]");
            nombreClientes[i]=entrarDatos.nextLine();
            System.out.println("Cual son los apellidos "+"["+i+"]");
            apellidosClientes[i]=entrarDatos.nextLine();
            } 
         
         while(visualizar) {
           System.out.println("Nombres: "+nombreClientes[j] );
           System.out.println("Apellido: "+apellidosClientes[j] );
           
            if (1==j){
               visualizar=false;
             }
            j++;
         }   
     
    }}
